250  uint32_t num_trx = b.transactions.size();
251

253  trx_size += fc::raw::pack_size(trx);
254  }

256 
257  for (auto bucket : _tracked_buckets) {
258  auto open = fc::time_point_sec(
259  (db.head_block_time().sec_since_epoch() / bucket) *

261  auto itr = bucket_idx.find(boost::make_tuple(bucket, open));
262

264  _current_buckets.insert(

268  bo.blocks = 1;
269  }).id);
270 
271  if (_maximum_history_per_bucket_size > 0) {
272  try {

275  safe<uint32_t>(bucket) *
