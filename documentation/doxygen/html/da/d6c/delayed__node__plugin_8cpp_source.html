75  void delayed_node_plugin::sync_with_trusted_node() {
76  auto &db = database();
77  uint32_t synced_blocks = 0;
78  uint32_t pass_count = 0;
79  while (true) {
80  steemit::chain::dynamic_global_property_object remote_dpo = my->database_api->get_dynamic_global_properties();
81  if (remote_dpo.last_irreversible_block_num <=
82  db.head_block_num()) {
83  if (remote_dpo.last_irreversible_block_num <
84  db.head_block_num()) {
85  wlog("Trusted node seems to be behind delayed node");
86  }
87  if (synced_blocks > 1) {
88  ilog("Delayed node finished syncing ${n} blocks in ${k} passes", ("n", synced_blocks)("k", pass_count));
89  }
90  break;
91  }
92  pass_count++;
93  while (remote_dpo.last_irreversible_block_num >
94  db.head_block_num()) {
95  fc::optional<steemit::chain::signed_block> block = my->database_api->get_block(
96  db.head_block_num() + 1);
97  FC_ASSERT(block, "Trusted node claims it has blocks it doesn't actually have.");
98  ilog("Pushing block #${n}", ("n", block->block_num()));
99  db.push_block(*block);
100  synced_blocks++;
101  }
102  }
103  }

105  void delayed_node_plugin::mainloop() {
106  while (true) {
107  try {
108  fc::usleep(fc::microseconds(296645)); // wake up a little over 3Hz
109 
110  if (my->last_received_remote_head ==
111  my->last_processed_remote_head) {
112  continue;
113  }
114 
115  sync_with_trusted_node();
116  my->last_processed_remote_head = my->last_received_remote_head;
117  }
118  catch (const fc::exception &e) {
119  elog("Error during connection: ${e}", ("e", e.to_detail_string()));
120  }
121  }
122  }

124  void delayed_node_plugin::plugin_startup() {
