1120  fc::bigint entropy1(sha_entropy1.data(), sha_entropy1.data_size());
1121  fc::bigint entropy2(sha_entropy2.data(), sha_entropy2.data_size());
1122  fc::bigint entropy(entropy1);
1123  entropy <<= 8 * sha_entropy1.data_size();
1124  entropy += entropy2;

1126

1129  entropy /= graphene::words::word_list_size;
