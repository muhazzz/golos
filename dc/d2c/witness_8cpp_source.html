85 void witness_plugin::plugin_initialize(const boost::program_options::variables_map &options) {
86  try {
87  ilog("witness plugin: plugin_initialize() begin");
88  _options = &options;
89  LOAD_VALUE_SET(options, "witness", _witnesses, string)
90  edump((_witnesses));
91 
92  if (options.count("miner")) {
93 
94  const vector<string> miner_to_wif_pair_strings = options["miner"].as<vector<string>>();
95  for (auto p : miner_to_wif_pair_strings) {
96  auto m = steemit::app::dejsonify<pair<string, string>>(p);
97  idump((m));
98

100  FC_ASSERT(private_key.valid(), "unable to parse private key");
101  _private_keys[private_key->get_public_key()] = *private_key;
102  _miners[m.first] = private_key->get_public_key();

104  }
105

148  }

151  } FC_LOG_AND_RETHROW()

155  try {
156  ilog("witness plugin: plugin_startup() begin");
157  chain::database &d = database();
158  //Start NTP time client
159  graphene::time::now();
160

192 void witness_plugin::schedule_production_loop() {
193  //Schedule for the next second's tick regardless of chain state
194  // If we would wait less than 50ms, wait for the whole second.
195  fc::time_point ntp_now = graphene::time::now();
196  fc::time_point fc_now = fc::time_point::now();
197  int64_t time_to_next_second =
198  1000000 - (ntp_now.time_since_epoch().count() % 1000000);
199  if (time_to_next_second < 50000) { // we must sleep for at least 50ms
200  time_to_next_second += 1000000;
201  }
202 
203  fc::time_point next_wakeup(fc_now + fc::microseconds(time_to_next_second));
204 
205  //wdump( (now.time_since_epoch().count())(next_wakeup.time_since_epoch().count()) );
206  _block_production_task = fc::schedule([this] { block_production_loop(); },
207  next_wakeup, "Witness Block Production");
208 }

210 block_production_condition::block_production_condition_enum witness_plugin::block_production_loop() {
211  if (fc::time_point::now() < fc::time_point(STEEMIT_GENESIS_TIME)) {
212  wlog("waiting until genesis time to produce block: ${t}", ("t", STEEMIT_GENESIS_TIME));
213  schedule_production_loop();
214  return block_production_condition::wait_for_genesis;
215  }
216 
217  block_production_condition::block_production_condition_enum result;
218  fc::mutable_variant_object capture;
219  try {
220  result = maybe_produce_block(capture);
221  }
222  catch (const fc::canceled_exception &) {
223  //We're trying to exit. Go ahead and let this one out.
224  throw;
225  }
226  catch (const steemit::chain::unknown_hardfork_exception &e) {
227  // Hit a hardfork that the current node know nothing about, stop production and inform user
228  elog("${e}\nNode may be out of date...", ("e", e.to_detail_string()));
229  throw;
230  }
231  catch (const fc::exception &e) {
232  elog("Got exception while generating block:\n${e}", ("e", e.to_detail_string()));
233  result = block_production_condition::exception_producing_block;
234  }
235 
236  switch (result) {
237  case block_production_condition::produced:
238  ilog("Generated block #${n} with timestamp ${t} at time ${c} by ${w}", (capture));
239  break;
240  case block_production_condition::not_synced:
241  //ilog("Not producing block because production is disabled until we receive a recent block (see: --enable-stale-production)");
242  break;
243  case block_production_condition::not_my_turn:
244  //ilog("Not producing block because it isn't my turn");
245  break;
246  case block_production_condition::not_time_yet:
247  // ilog("Not producing block because slot has not yet arrived");
248  break;
249  case block_production_condition::no_private_key:
250  ilog("Not producing block for ${scheduled_witness} because I don't have the private key for ${scheduled_key}", (capture));
251  break;
252  case block_production_condition::low_participation:
253  elog("Not producing block because node appears to be on a minority fork with only ${pct}% witness participation", (capture));
254  break;
255  case block_production_condition::lag:
256  elog("Not producing block because node didn't wake up within 500ms of the slot time.");
257  break;
258  case block_production_condition::consecutive:
259  elog("Not producing block because the last block was generated by the same witness.\nThis node is probably disconnected from the network so block production has been disabled.\nDisable this check with --allow-consecutive option.");
260  break;
261  case block_production_condition::exception_producing_block:
262  elog("Failure when producing block with no transactions");
263  break;
264  case block_production_condition::wait_for_genesis:
265  break;
266  }
267 
268  schedule_production_loop();
269  return result;
270 }

272 block_production_condition::block_production_condition_enum witness_plugin::maybe_produce_block(fc::mutable_variant_object &capture) {
273  chain::database &db = database();
274  fc::time_point now_fine = graphene::time::now();
275  fc::time_point_sec now = now_fine + fc::microseconds(500000);
276 
277  // If the next block production opportunity is in the present or future, we're synced.
278  if (!_production_enabled) {
279  if (db.get_slot_time(1) >= now) {
280  _production_enabled = true;
281  } else {
282  return block_production_condition::not_synced;
283  }
284  }
285 
286  // is anyone scheduled to produce now or one second in the future?
287  uint32_t slot = db.get_slot_at_time(now);
288  if (slot == 0) {
289  capture("next_time", db.get_slot_time(1));
290  return block_production_condition::not_time_yet;
291  }
292 
293  //
294  // this assert should not fail, because now <= db.head_block_time()
295  // should have resulted in slot == 0.
296  //
297  // if this assert triggers, there is a serious bug in get_slot_at_time()
298  // which would result in allowing a later block to have a timestamp
299  // less than or equal to the previous block
300  //
301  assert(now > db.head_block_time());
302 
303  string scheduled_witness = db.get_scheduled_witness(slot);
304  // we must control the witness scheduled to produce the next block.
305  if (_witnesses.find(scheduled_witness) == _witnesses.end()) {
306  capture("scheduled_witness", scheduled_witness);
307  return block_production_condition::not_my_turn;
308  }
309 
310  const auto &witness_by_name = db.get_index<chain::witness_index>().indices().get<chain::by_name>();
311  auto itr = witness_by_name.find(scheduled_witness);
312 
313  fc::time_point_sec scheduled_time = db.get_slot_time(slot);
314  steemit::protocol::public_key_type scheduled_key = itr->signing_key;
315  auto private_key_itr = _private_keys.find(scheduled_key);
316 
317  if (private_key_itr == _private_keys.end()) {
318  capture("scheduled_witness", scheduled_witness);
319  capture("scheduled_key", scheduled_key);
320  return block_production_condition::no_private_key;
321  }
322 
323  uint32_t prate = db.witness_participation_rate();
324  if (prate < _required_witness_participation) {
325  capture("pct", uint32_t(100 * uint64_t(prate) / STEEMIT_1_PERCENT));
326  return block_production_condition::low_participation;
327  }
328 
329  if (llabs((scheduled_time - now).count()) > fc::milliseconds(500).count()) {
330  capture("scheduled_time", scheduled_time)("now", now);
331  return block_production_condition::lag;
332  }
333 
334  int retry = 0;
335  do {
336  try {
337  auto block = db.generate_block(
338  scheduled_time,
339  scheduled_witness,
340  private_key_itr->second,
341  _production_skip_flags
342  );
343  capture("n", block.block_num())("t", block.timestamp)("c", now)("w", scheduled_witness);
344  fc::async([this, block]() { p2p_node().broadcast(graphene::net::block_message(block)); });
345 
346  return block_production_condition::produced;
347  }
348  catch (fc::exception &e) {
349  elog("${e}", ("e", e.to_detail_string()));
350  elog("Clearing pending transactions and attempting again");
351  db.clear_pending();
352  retry++;
353  }
354  } while (retry < 2);
355 
356  return block_production_condition::exception_producing_block;
357 }

373 void witness_plugin::on_applied_block(const steemit::protocol::signed_block &b) {
374  try {
375  if (!_mining_threads || _miners.size() == 0) {
376  return;
377  }
378  chain::database &db = database();
379 
380  const auto &dgp = db.get_dynamic_global_properties();
381  double hps = (_total_hashes * 1000000) /
382  (fc::time_point::now() - _hash_start_time).count();
383  uint64_t i_hps = uint64_t(hps + 0.5);
384 
385  uint32_t summary_target = db.get_pow_summary_target();
386 
387  double target = fc::sha256::inverse_approx_log_32_double(summary_target);
388  static const double max_target = std::ldexp(1.0, 256);
389 
390  double seconds_needed = 0.0;
391  if (i_hps > 0) {
392  double hashes_needed = max_target / target;
393  seconds_needed = hashes_needed / i_hps;
394  }
395 
396  uint64_t minutes_needed = uint64_t(seconds_needed / 60.0 + 0.5);
397 
398  fc::sha256 hash_target;
399  hash_target.set_to_inverse_approx_log_32(summary_target);
400 
401  if (_total_hashes > 0)
402  ilog("hash rate: ${x} hps target: ${t} queue: ${l} estimated time to produce: ${m} minutes",
403  ("x", i_hps)("t", hash_target.str())("m", minutes_needed)("l", dgp.num_pow_witnesses)
404  );
405 
406 
407  _head_block_num = b.block_num();
408  _head_block_id = b.id();
410 
411  for (const auto &miner : _miners) {
412  const auto *w = db.find_witness(miner.first);
413  if (!w || w->pow_worker == 0) {
414  auto miner_pub_key = miner.second; //a.active.key_auths.begin()->first;
415  auto priv_key_itr = _private_keys.find(miner_pub_key);
416  if (priv_key_itr == _private_keys.end()) {
417  continue;
418  }
419 
420  auto miner_priv_key = priv_key_itr->second;
421  start_mining(miner_pub_key, priv_key_itr->second, miner.first, b);
422  break;
423  } else {
424  // ilog( "Skipping miner ${m} because it is already scheduled to produce a block", ("m",miner) );
425  }
426  } // for miner in miners
427 
428  } catch (const fc::exception &e) {
429  ilog("exception thrown while attempting to mine");
430  }
431 }

433 void witness_plugin::start_mining(
434  const fc::ecc::public_key &pub,
435  const fc::ecc::private_key &pk,
436  const string &miner,
437  const steemit::protocol::signed_block &b) {
438  static uint64_t seed = fc::time_point::now().time_since_epoch().count();
439  static uint64_t start = fc::city_hash64((const char *)&seed, sizeof(seed));
440  chain::database &db = database();
441  auto head_block_num = b.block_num();
442  auto head_block_time = b.timestamp;
443  auto block_id = b.id();
444  fc::thread *mainthread = &fc::thread::current();
445  _total_hashes = 0;
446  _hash_start_time = fc::time_point::now();
447  auto stop = head_block_time + fc::seconds(STEEMIT_BLOCK_INTERVAL * 2);
448  uint32_t thread_num = 0;
449  uint32_t num_threads = _mining_threads;
450  uint32_t target = db.get_pow_summary_target();
451  const auto &acct_idx = db.get_index<chain::account_index>().indices().get<chain::by_name>();
452  auto acct_it = acct_idx.find(miner);
453  bool has_account = (acct_it != acct_idx.end());
454  bool has_hardfork_16 = db.has_hardfork(STEEMIT_HARDFORK_0_16__551);
455  for (auto &t : _thread_pool) {
456  thread_num++;
457  t->async([=]() {
458  if (has_hardfork_16) {
459  protocol::pow2_operation op;
460  protocol::equihash_pow work;
461  work.input.prev_block = block_id;
462  work.input.worker_account = miner;
463  work.input.nonce = start + thread_num;
464  op.props = _miner_prop_vote;
465 
466  while (true) {
467  if (graphene::time::nonblocking_now() > stop) {
468  // ilog( "stop mining due to time out, nonce: ${n}", ("n",op.nonce) );
469  return;
470  }
471  if (this->_head_block_num != head_block_num) {
472  // wlog( "stop mining due new block arrival, nonce: ${n}", ("n",op.nonce));
473  return;
474  }
475 
476  ++this->_total_hashes;
477  work.input.nonce += num_threads;
478  work.create(block_id, miner, work.input.nonce);
479 
480  if (work.proof.is_valid() && work.pow_summary < target) {
481  protocol::signed_transaction trx;
482  work.prev_block = this->_head_block_id;
483  op.work = work;
484  if (!has_account) {
485  op.new_owner_key = pub;
486  }
487  trx.operations.push_back(op);
488  trx.ref_block_num = head_block_num;
489  trx.ref_block_prefix = work.input.prev_block._hash[1];
490  trx.set_expiration(head_block_time +
491  STEEMIT_MAX_TIME_UNTIL_EXPIRATION);
492  trx.sign(pk, STEEMIT_CHAIN_ID);
493  ++this->_head_block_num;
494  mainthread->async([this, miner, trx]() {
495  try {
496  database().push_transaction(trx);
497  ilog("Broadcasting Proof of Work for ${miner}", ("miner", miner));
498  p2p_node().broadcast(graphene::net::trx_message(trx));
499  }
500  catch (const fc::exception &e) {
501  // wdump((e.to_detail_string()));
502  }
503  });
504  return;
505  }
506  }
507  } else // delete after hardfork 16
508  {
509  protocol::pow2_operation op;
510  protocol::pow2 work;
511  work.input.prev_block = block_id;
512  work.input.worker_account = miner;
513  work.input.nonce = start + thread_num;
514  op.props = _miner_prop_vote;
515  while (true) {
516  // if( ((op.nonce/num_threads) % 1000) == 0 ) idump((op.nonce));
517  if (graphene::time::nonblocking_now() > stop) {
518  // ilog( "stop mining due to time out, nonce: ${n}", ("n",op.nonce) );
519  return;
520  }
521  if (this->_head_block_num != head_block_num) {
522  // wlog( "stop mining due new block arrival, nonce: ${n}", ("n",op.nonce));
523  return;
524  }
525 
526  ++this->_total_hashes;
527  work.input.nonce += num_threads;
528  work.create(block_id, miner, work.input.nonce);
529  if (work.pow_summary < target) {
530  ++this->_head_block_num;
531  protocol::signed_transaction trx;
532  op.work = work;
533  if (!has_account) {
534  op.new_owner_key = pub;
535  }
536  trx.operations.push_back(op);
537  trx.ref_block_num = head_block_num;
538  trx.ref_block_prefix = work.input.prev_block._hash[1];
539  trx.set_expiration(head_block_time +
540  STEEMIT_MAX_TIME_UNTIL_EXPIRATION);
541  trx.sign(pk, STEEMIT_CHAIN_ID);
542  mainthread->async([this, miner, trx]() {
543  try {
544  database().push_transaction(trx);
545  ilog("Broadcasting Proof of Work for ${miner}", ("miner", miner));
546  p2p_node().broadcast(graphene::net::trx_message(trx));
547  }
548  catch (const fc::exception &e) {
549  // wdump((e.to_detail_string()));
550  }
551  });
552  return;
553  }
554  }
555  }
556  });
557  thread_num++;
558  }
