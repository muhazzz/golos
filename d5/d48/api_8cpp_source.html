49  bool login_api::login(const string &user, const string &password) {
50  idump((user)(password));
51  optional<api_access_info> acc = _ctx.app.get_api_access_info(user);
52  if (!acc.valid()) {
53  return false;
54  }
55  if (acc->password_hash_b64 != "*") {
56  std::string password_salt = fc::base64_decode(acc->password_salt_b64);
57  std::string acc_password_hash = fc::base64_decode(acc->password_hash_b64);
58 
59  fc::sha256 hash_obj = fc::sha256::hash(
60  password + password_salt);
61  if (hash_obj.data_size() != acc_password_hash.length()) {
62  return false;
63  }
64  if (memcmp(hash_obj.data(), acc_password_hash.c_str(), hash_obj.data_size()) !=
65  0) {
66  return false;
67  }
68  }
69 
70  idump((acc->allowed_apis));
71  std::shared_ptr<api_session_data> session = _ctx.session.lock();
72  FC_ASSERT(session);
73 
74  std::map<std::string, fc::api_ptr> &_api_map = session->api_map;
75 
76  for (const std::string &api_name : acc->allowed_apis) {
77  auto it = _api_map.find(api_name);
78  if (it != _api_map.end()) {
79  wlog("known api: ${api}", ("api", api_name));
80  continue;
81  }
82  idump((api_name));
83  api_context new_ctx(_ctx.app, api_name, _ctx.session);
84  _api_map[api_name] = _ctx.app.create_api_by_name(new_ctx);
85  }
86  return true;
87  }

89  fc::api_ptr login_api::get_api_by_name(const string &api_name) const {
90  std::shared_ptr<api_session_data> session = _ctx.session.lock();
91  FC_ASSERT(session);
92 
93  const std::map<std::string, fc::api_ptr> &_api_map = session->api_map;
94  auto it = _api_map.find(api_name);
95  if (it == _api_map.end()) {
96  wlog("unknown api: ${api}", ("api", api_name));
97  return fc::api_ptr();
98  }
99  if (it->second) {

102  FC_ASSERT(it->second != nullptr);

125  bool network_broadcast_api::check_max_block_age(int32_t max_block_age) {
126  return _app.chain_database()->with_read_lock([&]() {
127  if (max_block_age < 0) {
128  return false;
129  }
130 
131  fc::time_point_sec now = graphene::time::now();
132  std::shared_ptr<database> db = _app.chain_database();
133  const dynamic_global_property_object &dgpo = db->get_dynamic_global_properties();
134 
135  return (dgpo.time < now - fc::seconds(max_block_age));

143  void network_broadcast_api::on_applied_block(const signed_block &b) {
145  auto capture_this = shared_from_this();
146 
147  fc::async([this, capture_this, b]() {
148  int32_t block_num = int32_t(b.block_num());
149  if (_callbacks.size()) {
150  for (size_t trx_num = 0;
151  trx_num < b.transactions.size(); ++trx_num) {
152  const auto &trx = b.transactions[trx_num];
153  auto id = trx.id();
154  auto itr = _callbacks.find(id);
155  if (itr == _callbacks.end()) {
156  continue;
157  }
158  confirmation_callback callback = itr->second;
159  itr->second = [](variant) {};
160  callback(fc::variant(transaction_confirmation(id, block_num, int32_t(trx_num), false)));
161  }
162  }
163 
165  while (true) {
166  auto exp_it = _callbacks_expirations.begin();
167  if (exp_it == _callbacks_expirations.end()) {
168  break;
169  }
170  if (exp_it->first >= b.timestamp) {
171  break;
172  }
173  for (const transaction_id_type &txid : exp_it->second) {
174  auto cb_it = _callbacks.find(txid);
175  // If it's empty, that means the transaction has been confirmed and has been deleted by the above check.
176  if (cb_it == _callbacks.end()) {
177  continue;
178  }
179 
180  confirmation_callback callback = cb_it->second;
181  transaction_id_type txid_byval = txid; // can't pass in by reference as it's going to be deleted
182  callback(fc::variant(transaction_confirmation{
183  txid_byval, block_num, -1, true
184  }));
185 
186  _callbacks.erase(cb_it);
187  }
188  _callbacks_expirations.erase(exp_it);
189  }
190  });
191 
192  }

194  void network_broadcast_api::broadcast_transaction(const signed_transaction &trx) {
195  trx.validate();
196 
197  if (_app._read_only) {
198  FC_ASSERT(_app._remote_net_api, "Write node RPC not configured properly or non connected.");
199  (*_app._remote_net_api)->broadcast_transaction(trx);
200  } else {
201  FC_ASSERT(!check_max_block_age(_max_block_age));
202  _app.chain_database()->push_transaction(trx);
