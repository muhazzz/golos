175  void account_history_plugin_impl::on_operation(const operation_notification &note) {
176  flat_set<account_name_type> impacted;
177  steemit::chain::database &db = database();
178 
179  const operation_object *new_obj = nullptr;
180  app::operation_get_impacted_accounts(note.op, impacted);
181 
182  for (const auto &item : impacted) {
183  auto itr = _tracked_accounts.lower_bound(item);
184  if (!_tracked_accounts.size() ||
185  (itr != _tracked_accounts.end() && itr->first <= item &&
186  item <= itr->second)) {
187  if (_filter_content) {
188  note.op.visit(operation_visitor_filter(db, note, new_obj, item));
189  } else {
190  note.op.visit(operation_visitor(db, note, new_obj, item));
191  }
192  }
193  }
194  }
