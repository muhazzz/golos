70  void private_message_evaluator::do_apply(const private_message_operation &pm) {
71  database &d = db();
72 
73  const flat_map<string, string> &tracked_accounts = _plugin->my->_tracked_accounts;
74 
75  auto to_itr = tracked_accounts.lower_bound(pm.to);
76  auto from_itr = tracked_accounts.lower_bound(pm.from);
77 
78  FC_ASSERT(pm.from != pm.to);
79  FC_ASSERT(pm.from_memo_key != pm.to_memo_key);
80  FC_ASSERT(pm.sent_time != 0);
81  FC_ASSERT(pm.encrypted_message.size() >= 32);
82 
83  if (!tracked_accounts.size() ||
84  (to_itr != tracked_accounts.end() && pm.to >= to_itr->first &&
85  pm.to <= to_itr->second) ||
86  (from_itr != tracked_accounts.end() &&
87  pm.from >= from_itr->first && pm.from <= from_itr->second)) {
88  d.create<message_object>([&](message_object &pmo) {
89  pmo.from = pm.from;
90  pmo.to = pm.to;
91  pmo.from_memo_key = pm.from_memo_key;
92  pmo.to_memo_key = pm.to_memo_key;
93  pmo.checksum = pm.checksum;
94  pmo.sent_time = pm.sent_time;
95  pmo.receive_time = d.head_block_time();
96  pmo.encrypted_message.resize(pm.encrypted_message.size());
97  std::copy(pm.encrypted_message.begin(), pm.encrypted_message.end(), pmo.encrypted_message.begin());
98  });
99  }
100  }

134  vector<message_api_obj> private_message_api::get_inbox(string to, time_point newest, uint16_t limit) const {
135  FC_ASSERT(limit <= 100);
136  vector<message_api_obj> result;
137  const auto &idx = _app->chain_database()->get_index<message_index>().indices().get<by_to_date>();
138  auto itr = idx.lower_bound(std::make_tuple(to, newest));
139  while (itr != idx.end() && limit && itr->to == to) {
140  result.push_back(*itr);
141  ++itr;
142  --limit;
143  }
144 
145  return result;
