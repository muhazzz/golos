174  void account_by_key_plugin_impl::update_key_lookup(const account_authority_object &a) {
175  auto &db = database();
176  flat_set<public_key_type> new_keys;
177 
178  // Construct the set of keys in the account's authority
179  for (const auto &item : a.owner.key_auths) {
180  new_keys.insert(item.first);
181  }
182  for (const auto &item : a.active.key_auths) {
183  new_keys.insert(item.first);
184  }
185  for (const auto &item : a.posting.key_auths) {
186  new_keys.insert(item.first);
187  }
188 
189  // For each key that needs a lookup
190  for (const auto &key : new_keys) {
191  // If the key was not in the authority, add it to the lookup
192  if (cached_keys.find(key) == cached_keys.end()) {
193  auto lookup_itr = db.find<key_lookup_object, by_key>(std::make_tuple(key, a.account));
194 
195  if (lookup_itr == nullptr) {
196  db.create<key_lookup_object>([&](key_lookup_object &o) {
197  o.key = key;
198  o.account = a.account;
199  });
200  }
201  } else {
202  // If the key was already in the auths, remove it from the set so we don't delete it
203  cached_keys.erase(key);
204  }
205  }
206 
207  // Loop over the keys that were in authority but are no longer and remove them from the lookup
208  for (const auto &key : cached_keys) {
209  auto lookup_itr = db.find<key_lookup_object, by_key>(std::make_tuple(key, a.account));
210 
211  if (lookup_itr != nullptr) {
212  db.remove(*lookup_itr);
213  }
214  }
215 
216  cached_keys.clear();
217  }
