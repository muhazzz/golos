627 
628  if (!(skip & skip_fork_db)) {
629  shared_ptr<fork_item> new_head = _fork_db.push_block(new_block);
630  //If the head block from the longest chain does not build off of the current head, we need to switch forks.
631  if (new_head->data.previous != head_block_id()) {
632  //If the newly pushed block is the same height as head, we get head back in new_head
633  //Only switch forks if new_head is actually higher than head
634  if (new_head->data.block_num() > head_block_num()) {
635  // wlog( "Switching to fork: ${id}", ("id",new_head->data.id()) );
636  auto branches = _fork_db.fetch_branch_from(new_head->data.id(), head_block_id());
637 
638  // pop blocks until we hit the forked block
639  while (head_block_id() !=
640  branches.second.back()->data.previous) {
641  pop_block();
642  }
643 
644  // push all blocks on the new fork
645  for (auto ritr = branches.first.rbegin();
646  ritr != branches.first.rend(); ++ritr) {
647  // ilog( "pushing blocks from fork ${n} ${id}", ("n",(*ritr)->data.block_num())("id",(*ritr)->data.id()) );
648  optional<fc::exception> except;
649  try {
650  auto session = start_undo_session(true);
651  apply_block((*ritr)->data, skip);
652  session.push();
653  }
654  catch (const fc::exception &e) {
655  except = e;
656  }
657  if (except) {
658  // wlog( "exception thrown while switching forks ${e}", ("e",except->to_detail_string() ) );
659  // remove the rest of branches.first from the fork_db, those blocks are invalid
660  while (ritr != branches.first.rend()) {
661  _fork_db.remove((*ritr)->data.id());
662  ++ritr;
663  }
664  _fork_db.set_head(branches.second.front());
665 
666  // pop all blocks from the bad fork
667  while (head_block_id() !=
668  branches.second.back()->data.previous) {
669  pop_block();
670  }
671 
672  // restore all blocks from the good fork
673  for (auto ritr = branches.second.rbegin();
674  ritr !=
675  branches.second.rend(); ++ritr) {
676  auto session = start_undo_session(true);
677  apply_block((*ritr)->data, skip);
678  session.push();
679  }
680  throw *except;
681  }
682  }
683  return true;
684  } else {
685  return false;
686  }
687  }
688  }
689 
690  try {
691  auto session = start_undo_session(true);
692  apply_block(new_block, skip);
693  session.push();
694  }
695  catch (const fc::exception &e) {
696  elog("Failed to push new block:\n${e}", ("e", e.to_detail_string()));
697  _fork_db.remove(new_block.id());
698  throw;
699  }
700 
701  return false;
702  } FC_CAPTURE_AND_RETHROW()

737  void database::_push_transaction(const signed_transaction &trx) {
738  // If this is the first transaction pushed after applying a block, start a new undo session.
739  // This allows us to quickly rewind to the clean state of the head block, in case a new block arrives.
740  if (!_pending_tx_session.valid()) {
741  _pending_tx_session = start_undo_session(true);
742  }
743 
744  // Create a temporary undo session as a child of _pending_tx_session.
745  // The temporary session will be discarded by the destructor if
746  // _apply_transaction fails. If we make it to merge(), we
747  // apply the changes.
748 
749  auto temp_session = start_undo_session(true);
750  _apply_transaction(trx);
751  _pending_tx.push_back(trx);
752 
753  notify_changed_objects();
754  // The transaction applied successfully. Merge its changes into the pending block session.
755  temp_session.squash();
756 
757  // notify anyone listening to pending transactions
758  notify_on_pending_transaction(trx);
759  }

778  signed_block database::_generate_block(
779  fc::time_point_sec when,
780  const account_name_type &witness_owner,
781  const fc::ecc::private_key &block_signing_private_key
782  ) {
783  uint32_t skip = get_node_properties().skip_flags;
784  uint32_t slot_num = get_slot_at_time(when);
785  FC_ASSERT(slot_num > 0);
786  string scheduled_witness = get_scheduled_witness(slot_num);
787  FC_ASSERT(scheduled_witness == witness_owner);
788 
789  const auto &witness_obj = get_witness(witness_owner);
790 
791  if (!(skip & skip_witness_signature))
792  FC_ASSERT(witness_obj.signing_key ==
793  block_signing_private_key.get_public_key());
794 
795  static const size_t max_block_header_size =
796  fc::raw::pack_size(signed_block_header()) + 4;
797  auto maximum_block_size = get_dynamic_global_properties().maximum_block_size; //STEEMIT_MAX_BLOCK_SIZE;
798  size_t total_block_size = max_block_header_size;
799 
800  signed_block pending_block;
801 
802  with_write_lock([&]() {
803  //
804  // The following code throws away existing pending_tx_session and
805  // rebuilds it by re-applying pending transactions.
806  //
807  // This rebuild is necessary because pending transactions' validity
808  // and semantics may have changed since they were received, because
809  // time-based semantics are evaluated based on the current block
810  // time. These changes can only be reflected in the database when
811  // the value of the "when" variable is known, which means we need to
812  // re-apply pending transactions in this method.
813  //
814  _pending_tx_session.reset();
815  _pending_tx_session = start_undo_session(true);
816 
817  uint64_t postponed_tx_count = 0;
818  // pop pending state (reset to head block state)
819  for (const signed_transaction &tx : _pending_tx) {
820  // Only include transactions that have not expired yet for currently generating block,
821  // this should clear problem transactions and allow block production to continue
822 
823  if (tx.expiration < when) {
824  continue;
825  }
826 
827  uint64_t new_total_size =
828  total_block_size + fc::raw::pack_size(tx);
829 
830  // postpone transaction if it would make block too big
831  if (new_total_size >= maximum_block_size) {
832  postponed_tx_count++;
833  continue;
834  }
835 
836  try {
837  auto temp_session = start_undo_session(true);
838  _apply_transaction(tx);
839  temp_session.squash();
840 
841  total_block_size += fc::raw::pack_size(tx);
842  pending_block.transactions.push_back(tx);
843  }
844  catch (const fc::exception &e) {
845  // Do nothing, transaction will not be re-applied
846  //wlog( "Transaction was not processed while generating block due to ${e}", ("e", e) );
847  //wlog( "The transaction was ${t}", ("t", tx) );
848  }
849  }
850  if (postponed_tx_count > 0) {
851  wlog("Postponed ${n} transactions due to block size limit", ("n", postponed_tx_count));
852  }
853 
854  _pending_tx_session.reset();
855  });
856 
857  // We have temporarily broken the invariant that
858  // _pending_tx_session is the result of applying _pending_tx, as
859  // _pending_tx now consists of the set of postponed transactions.
860  // However, the push_block() call below will re-create the
861  // _pending_tx_session.
862 
863  pending_block.previous = head_block_id();
864  pending_block.timestamp = when;
865  pending_block.transaction_merkle_root = pending_block.calculate_merkle_root();
866  pending_block.witness = witness_owner;
867  if (has_hardfork(STEEMIT_HARDFORK_0_5__54)) {
868  const auto &witness = get_witness(witness_owner);
869 
870  if (witness.running_version != STEEMIT_BLOCKCHAIN_VERSION) {
871  pending_block.extensions.insert(block_header_extensions(STEEMIT_BLOCKCHAIN_VERSION));
872  }
873 
874  const auto &hfp = get_hardfork_property_object();
875 
876  if (hfp.current_hardfork_version <
877  STEEMIT_BLOCKCHAIN_HARDFORK_VERSION // Binary is newer hardfork than has been applied
878  && (witness.hardfork_version_vote !=
879  _hardfork_versions[hfp.last_hardfork + 1] ||
880  witness.hardfork_time_vote !=
881  _hardfork_times[hfp.last_hardfork +
882  1])) // Witness vote does not match binary configuration
883  {
884  // Make vote match binary configuration
885  pending_block.extensions.insert(block_header_extensions(hardfork_version_vote(_hardfork_versions[
886  hfp.last_hardfork + 1], _hardfork_times[
887  hfp.last_hardfork + 1])));
888  } else if (hfp.current_hardfork_version ==
889  STEEMIT_BLOCKCHAIN_HARDFORK_VERSION // Binary does not know of a new hardfork
890  && witness.hardfork_version_vote >
891  STEEMIT_BLOCKCHAIN_HARDFORK_VERSION) // Voting for hardfork in the future, that we do not know of...
892  {
893  // Make vote match binary configuration. This is vote to not apply the new hardfork.
894  pending_block.extensions.insert(block_header_extensions(hardfork_version_vote(_hardfork_versions[hfp.last_hardfork], _hardfork_times[hfp.last_hardfork])));
895  }
896  }
897 
898  if (!(skip & skip_witness_signature)) {
899  pending_block.sign(block_signing_private_key);
900  }
901 
902  // TODO: Move this to _push_block() so session is restored.
903  if (!(skip & skip_block_size_check)) {
904  FC_ASSERT(fc::raw::pack_size(pending_block) <=
905  STEEMIT_MAX_BLOCK_SIZE);
906  }
907 
908  push_block(pending_block, skip);
909 
910  return pending_block;
911  }
