40  std::string clean_name(const std::string &name) {
41  std::string result;
42  const static std::string prefix = "steemit::protocol::";
43  const static std::string suffix = "_operation";
44  // graphene::chain::.*_operation
45  if ((name.size() >= prefix.size() + suffix.size())
46  && (name.substr(0, prefix.size()) == prefix)
47  &&
48  (name.substr(name.size() - suffix.size(), suffix.size()) ==
49  suffix)
50  ) {
51  return name.substr(prefix.size(),
52  name.size() - prefix.size() - suffix.size());
53  }
