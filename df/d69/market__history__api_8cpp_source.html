95  auto itr = order_idx.lower_bound(price::max(SBD_SYMBOL, STEEM_SYMBOL));
96 
97  order_book result;
98

100  itr->sell_price.base.symbol == SBD_SYMBOL &&

102  order cur;
103  cur.price = itr->sell_price.base.to_real() /

105  cur.steem = (asset(itr->for_sale, SBD_SYMBOL) *
