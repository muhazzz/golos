34  statistics blockchain_statistics_api_impl::get_stats_for_interval(fc::time_point_sec start, fc::time_point_sec end) const {
35  statistics result;
36  const auto &bucket_itr = _app.chain_database()->get_index<bucket_index>().indices().get<by_bucket>();
37  const auto &sizes = _app.get_plugin<blockchain_statistics_plugin>(BLOCKCHAIN_STATISTICS_PLUGIN_NAME)->get_tracked_buckets();
38  auto size_itr = sizes.rbegin();
39  auto time = start;
40 
41  // This is a greedy algorithm, same as the ubiquitous "making change" problem.
42  // So long as the bucket sizes share a common denominator, the greedy solution
43  // has the same efficiency as the dynamic solution.
44  while (size_itr != sizes.rend() && time < end) {
45  auto itr = bucket_itr.find(boost::make_tuple(*size_itr, time));
46 
47  while (itr != bucket_itr.end() &&
48  itr->seconds == *size_itr &&
49  time + itr->seconds <= end) {
50  time += *size_itr;
51  result += *itr;
52  itr++;
53  }
54 
55  size_itr++;
56  }
57 
58  return result;
59  }
