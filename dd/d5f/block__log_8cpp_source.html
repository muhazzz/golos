88  * in relation to eachother.
89  *
90  * Block Log
91  * Exists Is New
92  * +------------+------------+
93  * Exists | Check | Delete |
94  * Index | Head | Index |
95  * File +------------+------------+
96  * Is New | Replay | Do |
97  * | Log | Nothing |
98  * +------------+------------+
99  *
100  * Checking the heads of the files has several conditions as well.
101  * - If they are the same, do nothing.
102  * - If the index file head is not in the log file, delete the index and replay.
103  * - If the index file head is in the log, but not up to date, replay from index head.
104  */
105  auto log_size = fc::file_size(my->block_file);
106  auto index_size = fc::file_size(my->index_file);
107 
108  if (log_size) {
109  ilog("Log is nonempty");
110  my->head = read_head();
111  my->head_id = my->head->id();
112 
113  if (index_size) {
