86  fc::optional<fc::ecc::private_key> &result,
87  const steemit::chain::public_key_type &pubkey,
88  const std::string &account_name
89  ) {
90  auto it = key_table.find(pubkey);
91  if (it != key_table.end()) {
92  result = it->second;
93  return;
94  }
95  fc::ecc::private_key gen_priv = fc::ecc::private_key::regenerate(fc::sha256::hash(
96  dev_key_prefix + account_name));
97  chain::public_key_type gen_pub = gen_priv.get_public_key();
98  key_table[gen_pub] = gen_priv;
99  if ((pubkey == steemit::chain::public_key_type()) ||
100  (gen_pub == pubkey)) {
101  result = gen_priv;
102  return;
103  }
104 
105  result.reset();
106  return;

130  void debug_node_api_impl::debug_mine(debug_mine_result &result, const debug_mine_args &args) {
131  std::shared_ptr<chain::database> db = app.chain_database();
132 
133  chain::pow2 work;
134  work.input.worker_account = args.worker_account;
135  work.input.prev_block = db->head_block_id();
136  get_plugin()->debug_mine_work(work, db->get_pow_summary_target());
137 
138  chain::pow2_operation op;
139  op.work = work;
140 
141  if (args.props.valid()) {
142  op.props = *(args.props);
143  } else {
144  op.props = db->get_witness_schedule_object().median_props;
145  }
146 
147  const auto &acct_idx = db->get_index<chain::account_index>().indices().get<chain::by_name>();
148  auto acct_it = acct_idx.find(args.worker_account);
149  auto acct_auth = db->find<chain::account_authority_object, chain::by_account>(args.worker_account);
150  bool has_account = (acct_it != acct_idx.end());
151 
152  fc::optional<fc::ecc::private_key> priv;
153  if (!has_account) {
154  // this copies logic from get_dev_key
155  priv = fc::ecc::private_key::regenerate(fc::sha256::hash(
156  key_storage.dev_key_prefix +
157  args.worker_account));
158  op.new_owner_key = priv->get_public_key();
159  } else {
160  chain::public_key_type pubkey;
161  if (acct_auth->active.key_auths.size() != 1) {
162  elog("debug_mine does not understand authority for miner account ${miner}", ("miner", args.worker_account));
163  }
164  FC_ASSERT(acct_auth->active.key_auths.size() == 1);
165  pubkey = acct_auth->active.key_auths.begin()->first;
166  key_storage.maybe_get_private_key(priv, pubkey, args.worker_account);
167  }
168  FC_ASSERT(priv.valid(), "debug_node_api does not know private key for miner account ${miner}", ("miner", args.worker_account));
169 
170  chain::signed_transaction tx;
171  tx.operations.push_back(op);
172  tx.ref_block_num = db->head_block_num();
173  tx.ref_block_prefix = work.input.prev_block._hash[1];
174  tx.set_expiration(db->head_block_time() +
175  STEEMIT_MAX_TIME_UNTIL_EXPIRATION);
176 
177  tx.sign(*priv, STEEMIT_CHAIN_ID);
178 
179  db->push_transaction(tx);
180  return;
181  }

183  uint32_t debug_node_api_impl::debug_push_blocks(const std::string &src_filename, uint32_t count, bool skip_validate_invariants) {
184  if (count == 0) {
185  return 0;
186  }
187 
188  std::shared_ptr<steemit::chain::database> db = app.chain_database();
189  fc::path src_path = fc::path(src_filename);
190  if (fc::is_directory(src_path)) {
191  ilog("Loading ${n} from block_log ${fn}", ("n", count)("fn", src_filename));
192  idump((src_filename)(count)(skip_validate_invariants));
193  steemit::chain::block_log log;
194  log.open(src_path);
195  uint32_t first_block = db->head_block_num() + 1;
196  uint32_t skip_flags = steemit::chain::database::skip_nothing;
197  if (skip_validate_invariants) {
198  skip_flags = skip_flags |
199  steemit::chain::database::skip_validate_invariants;
200  }
201  for (uint32_t i = 0; i < count; i++) {
202  //fc::optional< steemit::chain::signed_block > block = log.read_block( log.get_block_pos( first_block + i ) );
203  uint64_t block_pos = log.get_block_pos(
204  first_block + i);
205  if (block_pos == steemit::chain::block_log::npos) {
206  wlog("Block database ${fn} only contained ${i} of ${n} requested blocks", ("i", i)("n", count)("fn", src_filename));
207  return i;
208  }
209 
210  decltype(log.read_block(0)) result;
211 
212  try {
213  result = log.read_block(block_pos);
214  }
215  catch (const fc::exception &e) {
216  elog("Could not read block ${i} of ${n}", ("i", i)("n", count));
217  continue;
218  }
219 
220  try {
221  db->push_block(result.first, skip_flags);
222  }
223  catch (const fc::exception &e) {
224  elog("Got exception pushing block ${bn} : ${bid} (${i} of ${n})", ("bn", result.first.block_num())("bid", result.first.id())("i", i)("n", count));
225  elog("Exception backtrace: ${bt}", ("bt", e.to_detail_string()));
226  }
227  }
228  ilog("Completed loading block_database successfully");
229  return count;
230  }
231  return 0;
232  }
